package oohsms.assets {

	import flash.display.Sprite;
	import flash.display.MovieClip;
	import patota.display.safeRemoveChild;
	import patota.display.disposeChildren;
	import caurina.transitions.Tweener;
	public class Character extends Sprite {

		public var mc:MovieClip;
		public var finalX:Number;

		public function Character(char:MovieClip)
		{
			mc = char;

			if(!oohsms_main.debug)
			    safeRemoveChild(mc.bounds);

			addChild(mc);
		}

		public function hide():void
		{
		   Tweener.addTween(this,{time:.2, alpha: 0, onComplete: function():void{
		       safeRemoveChild(this);
		       dispose();
		   }})
		}

		public function get bounds():MovieClip
		{
		   return mc.bounds;
		}

		public function dispose():void
		{
		    if(mc) mc.stop();
		    safeRemoveChild(mc);
		    mc = null;
		   disposeChildren(this);
		}
	}

	}
